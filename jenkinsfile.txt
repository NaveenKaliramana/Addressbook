pipeline {
    agent any

    stages {
        stage ('Git Clone')  {
            steps  {
                git 'https://github.com/krnetworktraining1/Addressbook.git/'
            }
        }
        stage ('Compile Stage') {

            steps {
                    sh 'mvn compile'
            }
        }

        stage ('Testing Stage') {

            steps {
                    sh 'mvn test'
            }
        }


        stage ('Package Stage') {
            steps {
                    sh 'mvn package'
            }
        }
        
    }
}
